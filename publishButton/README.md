## Install

```bash
npm install --save button_soyhan
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'button_soyhan'
import 'button_soyhan/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
